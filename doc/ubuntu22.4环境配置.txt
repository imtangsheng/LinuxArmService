
1、配置网络：
network:
        ethernets:
                eth0:
                        dhcp4: false
                        addresses:
                                - 172.16.30.45/16   #网线ip
                                - 192.168.1.50/24   #雷达ip
                        routes:
                                - to: 0.0.0.0/0
                                  via: 172.16.30.254
                                  metric: 100
                                - to: 0.0.0.0/
                                  via: 192.168.1.1
                                  metric: 100
                        nameservers:
                                addresses:
                                        - 8.8.8.8
                                        - 144.144.144.144
        version: 2
		

2、关闭网络开机自检
systemctl mask systemd-networkd-wait-online.service

4、程序开机自启动
chmod 777 /lib/systemd/system/rc-local.service
vi /lib/systemd/system/rc-local.service
[Install]   
WantedBy=multi-user.target   
Alias=rc-local.service

vi /etc/rc.local
#! /bin/bash
if [ $(ps -elf | grep run.out | grep -v grep | wc -l) -eq 0 ];then
	cd /home/ubuntu/escan/program
	./update.out &
fi

chmod 777 /etc/rc.local
ln -s /lib/systemd/system/rc.local.service /etc/systemd/system/
